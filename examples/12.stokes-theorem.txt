with(linalg):

C:=[3*cos(t),3*sin(t),1-3*cos(t)-3*sin(t)]:

F:=[x^2*z,x*y^2,z^2]:
F:=simplify(eval(eval(eval(F,x=C[1]),y=C[2]),z=C[3])):

dC:=diff(C,t):
lineint:=int(innerprod(F,dC),t=0..2*Pi);


S:=[r*cos(t),r*sin(t),1-r*cos(t)-r*sin(t)]:

curlF:=curl([x^2*z,x*y^2,z^2],[x,y,z]):
curlF:=simplify(eval(eval(eval(curlF,x=S[1]),y=S[2]),z=S[3])):

dS:=simplify(crossprod(diff(S,r),diff(S,t))):
surfint:=int(int(innerprod(curlF,dS),r=0..3),t=0..2*Pi);
